
library IEEE;
use IEEE.STD_LOGIC_1164.ALL;



entity halfA is
port(A,B:in STD_LOGIC;
S,C:out STD_LOGIC);
end halfA;

architecture Behavioral of halfA is

begin
process(A,B)
begin
if A='0' and B='0' then
S<='0'; C<='0';
elsif A='0' and B='1' then
S<='1'; C<='0';
elsif A='1' and B='0' then
S<='1'; C<='0';
else 
S<='0'; C<='1';
end if;
end process;
end Behavioral;
